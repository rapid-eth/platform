## Install

```bash
npm install --save emblems-react-module
```

## Usage

```jsx
import React, { Component } from 'react'

import MyComponent from 'emblems-react-module'

class Example extends Component {
  render () {
    return (
      <MyComponent />
    )
  }
}
```
